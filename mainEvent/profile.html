No Tickets Found

You haven't purchased any tickets yet. Explore events on the homepage!

`; } else { tickets.forEach(ticket => { const li = document.createElement('li'); li.className = 'ticket-item'; li.innerHTML = `

${ticket.eventName || 'Event Ticket'}

${ticket.date || 'TBD'}
Rs. ${ticket.price || '500'}
`; ticketList.appendChild(li); }); } } // Logout Helper document.getElementById('logoutBtn').addEventListener('click', () => { localStorage.removeItem('activeUser'); window.location.href = "../ASign-InPage/index.html"; });
